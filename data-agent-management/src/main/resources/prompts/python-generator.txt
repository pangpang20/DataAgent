## 示例 1：分类数据对比（生成柱状图）

用户需求：统计各产品销售额

```python
import sys
import json
import traceback
import base64
from io import BytesIO
import pandas as pd
import subprocess

# 检查 matplotlib 是否可用，如果不可用则尝试动态安装
has_matplotlib = False
try:
    import matplotlib
    matplotlib.use('Agg')
    import matplotlib.pyplot as plt
    has_matplotlib = True
except ImportError:
    print("matplotlib not available, attempting to install...", file=sys.stderr)
    try:
        subprocess.check_call([sys.executable, "-m", "pip", "install", "matplotlib", "--quiet"])
        import matplotlib
        matplotlib.use('Agg')
        import matplotlib.pyplot as plt
        has_matplotlib = True
        print("matplotlib installed successfully", file=sys.stderr)
    except Exception as e:
        print(f"Failed to install matplotlib: {e}", file=sys.stderr)

def create_chart(data_df, x_col, y_col):
    try:
        fig, ax = plt.subplots(figsize=(10, 6), dpi=100)
        ax.bar(data_df[x_col].astype(str), data_df[y_col], color='#5584FF')
        ax.set_title(f'{{y_col}} by {{x_col}}', fontsize=12, fontweight='bold')
        ax.set_xlabel(x_col, fontsize=10)
        ax.set_ylabel(y_col, fontsize=10)
        plt.xticks(rotation=45, ha='right')
        ax.grid(axis='y', alpha=0.3)
        plt.tight_layout()

        buf = BytesIO()
        fig.savefig(buf, format='png', bbox_inches='tight', dpi=100)
        buf.seek(0)
        img_base64 = base64.b64encode(buf.read()).decode('utf-8')
        plt.close(fig)
        return img_base64
    except Exception as e:
        print(f"Chart creation failed: {{e}}", file=sys.stderr)
        return None

try:
    input_data = json.load(sys.stdin)
    df = pd.DataFrame(input_data)

    for col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='ignore')

    # 计算汇总统计
    result = {{
        "summary": {{
            "total_products": len(df),
            "total_revenue": df['TOTAL_REVENUE'].sum() if 'TOTAL_REVENUE' in df.columns else 0,
            "average_revenue": df['TOTAL_REVENUE'].mean() if 'TOTAL_REVENUE' in df.columns else 0
        }},
        "products": df.to_dict('records'),
        "chart_image": None
    }}

    if 'NAME' in df.columns and 'TOTAL_REVENUE' in df.columns and len(df) >= 2 and has_matplotlib:
        result["chart_image"] = create_chart(df, 'NAME', 'TOTAL_REVENUE')

    print(json.dumps(result, ensure_ascii=False))

except Exception:
    traceback.print_exc(file=sys.stderr)
    sys.exit(1)
```
